18 void HouseholderQR::compute_factorization() {
19  // For the intermediate calculations, we'll be working with RW.
20  // It is initialized to the rectangular matrix to be factored.
21  // At the end of this function, RW will contain the strict
22  // upper-triangular part of the matrix R (without the diagonal),
23  // and the complete scaled matrix of reflection vectors W, which is a
24  // lower-triangular matrix. The diagonal of R is stored separately in
25  // R_diag.
26 
27  assert(RW.rows() >= RW.cols());
28  assert(R_diag.size() == RW.cols());
29 
30  // Helper function to square a number
31  auto sq = [](double x) { return x * x; };
32 
33  for (size_t k = 0; k < RW.cols(); ++k) {
34  // Introduce a column vector x = A[k:M,k], it's the lower part of the
35  // k-th column of the matrix.
36  // First compute the norm of x:
37 
38  double sq_norm_x = 0;
39  for (size_t i = k; i < RW.rows(); ++i)
40  sq_norm_x += sq(RW(i, k));
41  double norm_x = std::sqrt(sq_norm_x);
42 
43  // x consists of two parts: its first element, x₀, and the rest, xₛ
44  // x = (x₀, xₛ)
45  // You can express the norm of x in terms of the norms of the two parts:
46  // ‖x‖² = x₀² + ‖xₛ‖²
47  double &x_0 = RW(k, k);
48 
49  // The goal of QR factorization is to introduce zeros below the diagonal
50  // in the R factor by transforming the vector x to a new vector that is
51  // all zero, except for the first component.
52  // Call this vector xₕ, the Householder reflection of x.
53  // Since the transformation has to be unitary (Q is a unitary matrix),
54  // this operation has to preserve the 2-norm. This means that the
55  // nonzero component of xₕ has to have the same 2-norm (energy) as x:
56  // xₕ = (±‖x‖, 0, ..., 0) = ±‖x‖·̅e₁
57  // where ̅e₁ is the first standard basis vector (1, 0, ..., 0).
58  //
59  // There are two (real) vectors that have the same energy as x in their
60  // first component, because the sign doesn't affect the energy.
61  // For numerical reasons, it's best to pick the sign opposite to the
62  // sign of the first component of x, so that x and xₕ are far apart.
63  //
64  // The reflector vector vₖ is the difference between the original
65  // vector x and its Householder reflection xₕ:
66  // vₖ = x - xₕ
67  // = x - (-sign(x₀)·‖x‖·̅e₁)
68  // = x + sign(x₀)·‖x‖·̅e₁
69  // = (x₀ + sign(x₀)·‖x‖, xₛ)
70  //
71  // Since vₖ will later be used to construct a projection matrix, it
72  // should be normalized.
73  // Computing the full norm is not necessary, since
74  // ‖vₖ‖² = (x₀ + sign(x₀)·‖x‖)² + ‖xₛ‖²
75  // = x₀² + 2·x₀·sign(x₀)·‖x‖ + ‖x‖² + ‖xₛ‖²
76  // = 2·x₀·sign(x₀)·‖x‖ + 2·‖x‖²
77  // = 2·|x₀|·‖x‖ + 2·‖x‖²
78  // ‖vₖ‖ = √2·√(|x₀|·‖x‖ + ‖x‖²)
79  //
80  // Normalize vₖ and call this uₖ:
81  // uₖ = vₖ / ‖vₖ‖
82  //
83  // For reasons that will become apparent in a moment, we'll keep the
84  // factor of √2 in our normalization of vₖ, call this vector wₖ:
85  // wₖ = √2·uₖ
86  // = √2·vₖ / ‖vₖ‖
87  // = vₖ / √(|x₀|·‖x‖ + ‖x‖²)
88  // Note how the sum only adds up numbers with the same sign. This
89  // prevents catastrophic cancelations.
90  //
91  // It is clear that if ‖x‖ = 0, this normalization will fail. In that
92  // case, we set uₖ = ̅e₁ or wₖ = √2·̅e₁.
93  //
94  // x will be overwritten by wₖ. The vector xₕ only has a single nonzero
95  // component. It is saved in the R_diag vector.
96 
97  if (norm_x >= std::numeric_limits<double>::min() * 2) {
98  double x_p = -std::copysign(norm_x, x_0); // -sign(x₀)·‖x‖
99  double v_0 = x_0 - x_p;
100  double norm_v_sq2 = std::sqrt(std::abs(x_0) * norm_x + sq_norm_x);
101 
102  // Overwrite x with vₖ:
103  x_0 = v_0;
104  // the other components of x (xₛ) are already equal to the bottom
105  // part of vₖ, so they don't have to be overwritten explicitly.
106 
107  // Then normalize x (= vₖ) to obtain wₖ:
108  for (size_t i = k; i < RW.rows(); ++i)
109  RW(i, k) /= norm_v_sq2;
110 
111  // Save the first component of xₕ:
112  R_diag(k) = x_p;
113  } else {
114  // Overwrite x with wₖ = √2·̅e₁:
115  x_0 = std::sqrt(2);
116  // the other components of x (xₛ) are already equal to zero, since
117  // ‖x‖ = 0.
118 
119  // Save the first component of xₕ:
120  R_diag(k) = 0;
121  }
122 
123  // Now that the reflection vector vₖ (wₖ) is known, the rest of the
124  // matrix A can be updated, that is A[k:m,k:n].
125  //
126  // The reflection was defined in terms of the reflection vector
127  // vₖ = x - xₕ. To reflect x onto xₕ, you can first project x onto the
128  // orthogonal complement of the space spanned by vₖ, call this
129  // projection xₚ, and then add the difference between xₚ and x to xₚ
130  // again, in order to reflect about the orthogonal complement of vₖ,
131  // rather than projecting onto it:
132  // xₕ = xₚ + (xₚ - x)
133  // = 2·xₚ - x
134  // The projection matrix onto the orthogonal complement of the space
135  // spanned by vₖ is:
136  // P⟂ = I - uₖ·uₖᵀ
137  // where I is the identity matrix, and uₖ the unit vector in the
138  // direction of vₖ, as defined above.
139  // This allows us to compute xₚ:
140  // xₚ = P⟂·x
141  // = I·x - uₖ·uₖᵀ·x
142  // = x - uₖ·uₖᵀ·x
143  // xₕ = 2·xₚ - x
144  // = 2·x - 2·uₖ·uₖᵀ·x - x
145  // = x - 2·uₖ·uₖᵀ·x
146  // Because of our choice of wₖ = √2·uₖ earlier, this simplifies to
147  // xₕ = x - wₖ·wₖᵀ·x
148  // The Householder reflector is thus the linear transformation given by
149  // H = I - wₖ·wₖᵀ
150  //
151  // The final step is to apply this reflector to the remaining part of
152  // the matrix A, i.e. A[k:m,k:n].
153  // The first column x = A[k:m,k] has already been updated, that's how
154  // the reflector was computed, so only the submatrix A[k+1:m,k:n] has
155  // to be updated.
156  // A'[k+1:m,k:n] = H·A[k+1:m,k:n]
157  // = A[k+1:m,k:n] - wₖ·wₖᵀ·A[k+1:m,k:n]
158  // This can be computed column-wise:
159  // aᵢ' = aᵢ - wₖ·wₖᵀ·aᵢ
160  // where aᵢ is the i-th column of A.
161 
162  for (size_t c = k + 1; c < RW.cols(); ++c) {
163  // Compute wₖᵀ·aᵢ
164  double dot_product = 0;
165  for (size_t r = k; r < RW.rows(); ++r)
166  dot_product += RW(r, k) * RW(r, c);
167  // Subtract wₖ·wₖᵀ·aᵢ
168  for (size_t r = k; r < RW.rows(); ++r)
169  RW(r, c) -= RW(r, k) * dot_product;
170  }
171  }
172  state = Factored;
173 }
