17 void LU::compute_factorization() {
18  // For the intermediate calculations, we'll be working with LU_.
19  // It is initialized to the square n×n matrix to be factored.
20 
21  assert(LU_.rows() == LU_.cols());
22 
23  // The goal of the LU factorization algorithm is to repeatedly apply
24  // transformations Lₖ to the matrix A to eventually end up with an upper-
25  // triangular matrix U:
26  //
27  // Lₙ⋯L₂L₁A = U
28  //
29  // The first transformation L₁ will introduce zeros below the diagonal in
30  // the first column of A, L₂ will introduce zeros below the diagonal in the
31  // second column of L₁A (while preserving the zeros introduced by L₁), and
32  // so on, until all elements below the diagonal are zero.
33 
34  // Loop over all columns of A:
35  for (size_t k = 0; k < LU_.cols(); ++k) {
36  // In the following comments, k = [1, n], because this is more intuitive
37  // and it follows the usual mathematical convention.
38  // In the code, however, array indices start at zero, so k = [0, n-1].
39 
40  // In order to introduce a zero in the i-th row and the k-th column of
41  // the matrix, subtract a multiple of the k-th row from the i-th row,
42  // using a scaling factor lᵢₖ such that
43  //
44  // A(i,k) - lᵢₖ·A(k,k) = 0
45  // lᵢₖ = A(i,k) / A(k,k)
46  //
47  // This is the typical Gaussian elimination algorithm.
48  // The element A(k,k) is often called the pivot.
49 
50  // The first update step (k=1) subtracts multiples of the first row from
51  // the rows below it.
52  // It can be represented as a lower-triangular matrix L₁:
53  // ┌ ┐
54  // │ 1 │
55  // │ -l₁₁ 1 │
56  // │ -l₂₁ 0 1 │
57  // │ -l₃₁ 0 0 1 │
58  // └ ┘
59  // Compute the product L₁A to verify this!
60  // You can see that the diagonal just preserves all rows, and the
61  // factors in the first columns of L₁ subtract a multiple of the first
62  // row from the other rows.
63  //
64  // The next step (k=2) follows exactly the same principle, but now
65  // subtracts multiples of the second row, resulting in a matrix L₂:
66  // ┌ ┐
67  // │ 1 │
68  // │ 0 1 │
69  // │ 0 -l₁₂ 1 │
70  // │ 0 -l₂₂ 0 1 │
71  // └ ┘
72  // This can be continued for all n columns of A.
73  //
74  // After applying matrices L₁ through Lₙ to A, it has been transformed
75  // into an upper-triangular matrix U:
76  //
77  // Lₙ⋯L₂L₁A = U
78  // A = L₁⁻¹L₂⁻¹⋯Lₙ⁻¹U
79  // A = LU
80  //
81  // Where
82  // L = L₁⁻¹L₂⁻¹⋯Lₙ⁻¹
83  //
84  // Luckily, inverting the matrices Lₖ is trivial, since they are sparse
85  // lower-triangular matrices with a determinant of 1, so their inverses
86  // will be equal to their adjugates.
87  // As an example, and without loss of generality, computing the adjugate
88  // of L₂ results in:
89  // ┌ ┐
90  // │ 1 │
91  // │ 0 1 │
92  // │ 0 l₁₂ 1 │
93  // │ 0 l₂₂ 0 1 │
94  // └ ┘
95  // This result is not immediately obvious, you have to write out some
96  // 3×3 determinants, but it's clear that many of them will be 0 or 1.
97  // Recall that the adjugate of a matrix is the transpose of the cofactor
98  // matrix.
99  //
100  // Finally, computing the product of all Lₖ⁻¹ factors is trivial as
101  // well, because of the structure of the factors. For example,
102  // L₁⁻¹L₂⁻¹ =
103  // ┌ ┐
104  // │ 1 │
105  // │ l₁₁ 1 │
106  // │ l₂₁ l₁₂ 1 │
107  // │ l₃₁ l₂₂ 0 1 │
108  // └ ┘
109  // In conclusion, we can just combine all factors lᵢₖ into the matrix L,
110  // without explicitly having to invert or multiply any matrices. Even
111  // the minus signs cancel out!
112 
113  // Note that by applying this transformation Lₖ to A, you'll always end
114  // up with zeros below the diagonal in the current column k, so there's
115  // no point in saving these zeros explicitly. Instead, this space is
116  // used to store the scaling factors lᵢₖ, i.e. the strict lower-
117  // triangular elements of L. The diagonal elements of L don't have to be
118  // stored either, because they're always 1.
119 
120  // Use the diagonal element as the pivot:
121  double pivot = LU_(k, k);
122 
123  // Compute the k-th column of L, the coefficients lᵢₖ:
124  for (size_t i = k + 1; i < LU_.rows(); ++i)
125  LU_(i, k) /= pivot;
126 
127  // Now update the rest of the matrix, we already (implicitly) introduced
128  // zeros below the diagonal in the k-th column, and we also stored the
129  // scaling factors for each row that determine Lₖ, but we haven't
130  // actually subtracted the multiples of the pivot row from the rest of
131  // the matrix yet, or in other words, we haven't multiplied the bottom
132  // right block of the matrix by Lₖ yet.
133  //
134  // Again, Lₖ has already been implicitly applied to the k-th column by
135  // setting all values below the diagonal to zero. Also the k-th row is
136  // unaffected by Lₖ because the k-th row of Lₖ is always equal to
137  // ̅eₖ = (0 ... 0 1 0 ... 0).
138  // This means only the rows and columns k+1 through n have to be
139  // updated.
140 
141  // Update the trailing submatrix A'(k+1:n,k+1:n) = LₖA(k+1:n,k+1:n):
142  for (size_t c = k + 1; c < LU_.cols(); ++c)
143  // Subtract lᵢₖ times the current pivot row A(k,:):
144  for (size_t i = k + 1; i < LU_.rows(); ++i)
145  // A'(i,c) = 1·A(i,c) - lᵢₖ·A(k,c)
146  LU_(i, c) -= LU_(i, k) * LU_(k, c);
147 
148  // We won't handle this here explicitly, but notice how the algorithm
149  // fails when the value of the pivot is zero (or very small), as this
150  // will cause a division by zero. When using IEEE 754 floating point
151  // numbers, this means that the factors lᵢₖ will overflow to ±∞,
152  // and during later calculations, infinities might be subtracted from
153  // eachother, resulting in many elements becoming NaN (Not a Number).
154  //
155  // Even ignoring the numerical issues with LU factorization, this is a
156  // huge dealbreaker: if a zero pivot is encountered anywhere during the
157  // factorization, it fails.
158  // Zero pivots occur even when the matrix is non-singular.
159  }
160  state = Factored;
161 }

163 void LU::back_subs(const Matrix &B, Matrix &X) const {
164  // Solve upper triangular system UX = B by solving each column of B as a
165  // vector system Uxᵢ = bᵢ
166  //
167  // ┌ ┐┌ ┐ ┌ ┐
168  // │ u₁₁ u₁₂ u₁₃ u₁₄ ││ x₁ᵢ │ │ b₁ᵢ │
169  // │ u₂₂ u₂₃ u₂₄ ││ x₂ᵢ │ = │ b₂ᵢ │
170  // │ u₃₃ u₃₄ ││ x₃ᵢ │ │ b₃ᵢ │
171  // │ u₄₄ ││ x₄ᵢ │ │ b₄ᵢ │
172  // └ ┘└ ┘ └ ┘
173  //
174  // b₄ᵢ = u₄₄·x₄ᵢ ⟺ x₄ᵢ = b₄ᵢ/u₄₄
175  // b₃ᵢ = u₃₃·x₃ᵢ + u₃₄·x₄ᵢ ⟺ x₃ᵢ = (b₃ᵢ - u₃₄·x₄ᵢ)/u₃₃
176  // b₂ᵢ = u₂₂·x₂ᵢ + u₂₃·x₃ᵢ + u₂₄·x₄ᵢ ⟺ x₂ᵢ = (b₂ᵢ - u₂₃·x₃ᵢ + u₂₄·x₄ᵢ)/u₂₂
177  // ...
178 
179  for (size_t i = 0; i < B.cols(); ++i) {
180  for (size_t r = LU_.rows(); r-- > 0;) {
181  X(r, i) = B(r, i);
182  for (size_t c = r + 1; c < LU_.cols(); ++c)
183  X(r, i) -= LU_(r, c) * X(c, i);
184  X(r, i) /= LU_(r, r);
185  }
186  }
187 }

189 void LU::forward_subs(const Matrix &B, Matrix &X) const {
190  // Solve lower triangular system LX = B by solving each column of B as a
191  // vector system Lxᵢ = bᵢ.
192  // The diagonal is always 1, due to the construction of the L matrix in the
193  // LU algorithm.
194  //
195  // ┌ ┐┌ ┐ ┌ ┐
196  // │ 1 ││ x₁ᵢ │ │ b₁ᵢ │
197  // │ l₂₁ 1 ││ x₂ᵢ │ = │ b₂ᵢ │
198  // │ l₃₁ l₃₂ 1 ││ x₃ᵢ │ │ b₃ᵢ │
199  // │ l₄₁ l₄₂ l₄₃ 1 ││ x₄ᵢ │ │ b₄ᵢ │
200  // └ ┘└ ┘ └ ┘
201  //
202  // b₁ᵢ = 1·x₁ᵢ ⟺ x₁ᵢ = b₁ᵢ
203  // b₂ᵢ = l₂₁·x₁ᵢ + 1·x₂ᵢ ⟺ x₂ᵢ = b₂ᵢ - l₂₁·x₁ᵢ
204  // b₃ᵢ = l₃₁·x₁ᵢ + l₃₂·x₂ᵢ + 1·x₃ᵢ ⟺ x₃ᵢ = b₃ᵢ - l₃₂·x₂ᵢ - l₃₁·x₁ᵢ
205  // ...
206 
207  for (size_t i = 0; i < B.cols(); ++i) {
208  for (size_t r = 0; r < LU_.rows(); ++r) {
209  X(r, i) = B(r, i);
210  for (size_t c = 0; c < r; ++c)
211  X(r, i) -= LU_(r, c) * X(c, i);
212  }
213  }
214 }

216 void LU::solve_inplace(Matrix &B) const {
217  // Solve the system AX = B, or LUX = B.
218  //
219  // Let UX = Z, and first solve LZ = B, which is a simple lower-triangular
220  // system of equations.
221  // Now that Z is known, solve UX = Z, which is a simple upper-triangular
222  // system of equations.
223  assert(is_factored());
224 
225  forward_subs(B, B); // overwrite B with Z
226  back_subs(B, B); // overwrite B (Z) with X
227 }
