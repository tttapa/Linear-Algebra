inline Matrix operator-(const Matrix &A, const Matrix &B) {
assert(A.rows() == B.rows());
assert(A.cols() == B.cols());
Matrix C(A.rows(), A.cols());
std::transform(A.begin(), A.end(), B.begin(), C.begin(),
std::minus<double>());
return C;
}
