412  Permutation p = identity_permutation(size());
413  auto &This = *this;
414  if (is_reversed()) {
415  // Count down
416  for (size_t i = size(); i-- > 0;)
417  if (i != This(i))
418  std::swap(p[i], p[This(i)]);
419  } else {
420  // Count up
