22 void RowPivotLU::compute_factorization() {
24  // For the intermediate calculations, we'll be working with LU.
25  // It is initialized to the square n×n matrix to be factored.
26 
27  assert(LU.rows() == LU.cols());
28  assert(P.size() == LU.rows());
29 
30  // The goal of the LU factorization algorithm is to repeatedly apply
31  // transformations Lₖ to the matrix A to eventually end up with an upper-
32  // triangular matrix U. When row pivoting is used, the rows of A are
33  // permuted using a permutation matrix P:
34  //
35  // Lₙ⋯L₂L₁PA = U
36  //
37  // The main steps of the algorithm are exactly the same as the original
38  // LU algorithm explained in LU.cpp, and will not be repeated here.
39  // The only difference is that instead of using the diagonal element as the
40  // pivot, rows are swapped so that the element with the largest magnitude
41  // ends up on the diagonal and can be used as the pivot.
42 
43  // Loop over all columns of A:
44  for (size_t k = 0; k < LU.cols(); ++k) {
45  // In the following comments, k = [1, n], because this is more intuitive
46  // and it follows the usual mathematical convention.
47  // In the code, however, array indices start at zero, so k = [0, n-1].
48 
49  // On each iteration, the largest element on or below the diagonal in
50  // the current (k-th) column will be used as the pivot.
51  // To this end, the k-th row and the row that contains the largest
52  // element are swapped, and the swapping is stored in the permutation
53  // matrix, so that it can later be undone, when solving systems of
54  // equations for example.
55 
56  // Find the largest element (in absolute value)
57  double max_elem = std::abs(LU(k, k));
58  size_t max_index = k;
59  for (size_t i = k + 1; i < LU.rows(); ++i) {
60  double abs_elem = std::abs(LU(i, k));
61  if (abs_elem > max_elem) {
62  max_elem = abs_elem;
63  max_index = i;
64  }
65  }
66 
67  // Select the index of the element that is largest in absolute value as
68  // the new pivot index.
69  // If this index is not the diagonal element, rows have to be swapped:
70  if (max_index != k) {
71  P(k) = max_index; // save the permutation
72  LU.swap_rows(k, max_index); // actually perfrom the permutation
73  }
74 
75  // The rest of the algorithm is identical to the one explained in
76  // LU.cpp.
77 
78  double pivot = LU(k, k);
79 
80  // Compute the k-th column of L, the coefficients lᵢₖ:
81  for (size_t i = k + 1; i < LU.rows(); ++i)
82  LU(i, k) /= pivot;
83 
84  // Update the trailing submatrix A'(k+1:n,k+1:n) = LₖA(k+1:n,k+1:n):
85  for (size_t c = k + 1; c < LU.cols(); ++c)
86  // Subtract lᵢₖ times the current pivot row A(k,:):
87  for (size_t i = k + 1; i < LU.rows(); ++i)
88  // A'(i,c) = 1·A(i,c) - lᵢₖ·A(k,c)
89  LU(i, c) -= LU(i, k) * LU(k, c);
90 
91  // Because of the row pivoting, zero pivots are no longer an issue,
92  // since the pivot is always chosen to be the largest possible element.
93  // When the matrix is singular, the algorithm will still fail, of
94  // course.
95  }
96  state = Factored;
97  valid_LU = true;
98  valid_P = true;
99 }

106 void RowPivotLU::back_subs(const Matrix &B, Matrix &X) const {
108  // Solve upper triangular system UX = B by solving each column of B as a
109  // vector system Uxᵢ = bᵢ
110  //
111  // ┌ ┐┌ ┐ ┌ ┐
112  // │ u₁₁ u₁₂ u₁₃ u₁₄ ││ x₁ᵢ │ │ b₁ᵢ │
113  // │ u₂₂ u₂₃ u₂₄ ││ x₂ᵢ │ = │ b₂ᵢ │
114  // │ u₃₃ u₃₄ ││ x₃ᵢ │ │ b₃ᵢ │
115  // │ u₄₄ ││ x₄ᵢ │ │ b₄ᵢ │
116  // └ ┘└ ┘ └ ┘
117  //
118  // b₄ᵢ = u₄₄·x₄ᵢ ⟺ x₄ᵢ = b₄ᵢ/u₄₄
119  // b₃ᵢ = u₃₃·x₃ᵢ + u₃₄·x₄ᵢ ⟺ x₃ᵢ = (b₃ᵢ - u₃₄·x₄ᵢ)/u₃₃
120  // b₂ᵢ = u₂₂·x₂ᵢ + u₂₃·x₃ᵢ + u₂₄·x₄ᵢ ⟺ x₂ᵢ = (b₂ᵢ - u₂₃·x₃ᵢ + u₂₄·x₄ᵢ)/u₂₂
121  // ...
122 
123  for (size_t i = 0; i < B.cols(); ++i) {
124  for (size_t r = LU.rows(); r-- > 0;) {
125  X(r, i) = B(r, i);
126  for (size_t c = r + 1; c < LU.cols(); ++c)
127  X(r, i) -= LU(r, c) * X(c, i);
128  X(r, i) /= LU(r, r);
129  }
130  }
131 }

138 void RowPivotLU::forward_subs(const Matrix &B, Matrix &X) const {
140  // Solve lower triangular system LX = B by solving each column of B as a
141  // vector system Lxᵢ = bᵢ.
142  // The diagonal is always 1, due to the construction of the L matrix in the
143  // LU algorithm.
144  //
145  // ┌ ┐┌ ┐ ┌ ┐
146  // │ 1 ││ x₁ᵢ │ │ b₁ᵢ │
147  // │ l₂₁ 1 ││ x₂ᵢ │ = │ b₂ᵢ │
148  // │ l₃₁ l₃₂ 1 ││ x₃ᵢ │ │ b₃ᵢ │
149  // │ l₄₁ l₄₂ l₄₃ 1 ││ x₄ᵢ │ │ b₄ᵢ │
150  // └ ┘└ ┘ └ ┘
151  //
152  // b₁ᵢ = 1·x₁ᵢ ⟺ x₁ᵢ = b₁ᵢ
153  // b₂ᵢ = l₂₁·x₁ᵢ + 1·x₂ᵢ ⟺ x₂ᵢ = b₂ᵢ - l₂₁·x₁ᵢ
154  // b₃ᵢ = l₃₁·x₁ᵢ + l₃₂·x₂ᵢ + 1·x₃ᵢ ⟺ x₃ᵢ = b₃ᵢ - l₃₂·x₂ᵢ - l₃₁·x₁ᵢ
155  // ...
156 
157  for (size_t i = 0; i < B.cols(); ++i) {
158  for (size_t r = 0; r < LU.rows(); ++r) {
159  X(r, i) = B(r, i);
160  for (size_t c = 0; c < r; ++c)
161  X(r, i) -= LU(r, c) * X(c, i);
162  }
163  }
164 }

171 void RowPivotLU::solve_inplace(Matrix &B) const {
173  // Solve the system AX = B, PAX = PB or LUX = PB.
174  //
175  // Let UX = Z, and first solve LZ = PB, which is a simple lower-triangular
176  // system of equations.
177  // Now that Z is known, solve UX = Z, which is a simple upper-triangular
178  // system of equations.
179  assert(is_factored());
180 
181  P.permute_rows(B);
182  forward_subs(B, B); // overwrite B with Z
183  back_subs(B, B); // overwrite B (Z) with X
184 }
